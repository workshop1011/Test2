@st.cache_data
def load_and_prep_data():
    try:
        base_dir = os.path.dirname(os.path.abspath(__file__))
        csv_dir = os.path.join(base_dir, 'CSV')
        
        cf_path = os.path.join(csv_dir, 'KOSPI_CF.csv')
        is_path = os.path.join(csv_dir, 'KOSPI_IS.csv')

        if not os.path.exists(cf_path) or not os.path.exists(is_path):
            csv_dir = os.path.join(os.getcwd(), 'CSV')
            cf_path = os.path.join(csv_dir, 'KOSPI_CF.csv')
            is_path = os.path.join(csv_dir, 'KOSPI_IS.csv')
            if not os.path.exists(cf_path): return None
            
        cf = pd.read_csv(cf_path)
        is_df = pd.read_csv(is_path)
        df = pd.merge(cf, is_df, on=['회사명', '회계년도'], how='inner')
        df.columns = df.columns.str.strip()
        
        def get_col(keywords):
            for kw in keywords:
                for col in df.columns:
                    if kw.replace(" ", "") in col.replace(" ", ""): return col
            return None

        c_ocf = get_col(['영업활동현금흐름', '영업활동으로인한현금흐름'])
        c_rev = get_col(['매출액', '매출'])
        c_op = get_col(['영업이익', '영업손익']) 
        c_ni = get_col(['당기순이익', '순이익'])

        if not all([c_ocf, c_rev, c_op, c_ni]):
            st.error("🚨 필수 재무 항목을 찾지 못했습니다.")
            return None

        def clean_finance_num(val):
            if pd.isna(val): return np.nan
            val_str = str(val).strip().replace(',', '')
            if val_str in ['-', '', '0']: return 0.0 
            try: return float(val_str)
            except: return np.nan

        for c in [c_ocf, c_rev, c_op, c_ni]:
            df[c] = df[c].apply(clean_finance_num)

        df['영업활동 현금(OCF)'] = df[c_ocf]
        df['매출액'] = df[c_rev]
        df['영업이익'] = df[c_op]
        df['영업이익률(%)'] = np.where(df['매출액'] > 0, (df['영업이익'] / df['매출액']) * 100, 0)
        df['이익의 질(OCF/NI)'] = np.where(df[c_ni] > 0, df['영업활동 현금(OCF)'] / df[c_ni], 0)
        
        return df.dropna(subset=['영업활동 현금(OCF)'])
        
    except Exception as e:
        st.error(f"🚨 로드 에러: {e}")
        return None

def calculate_metrics(y_true, y_pred):
    if len(y_true) < 2: return {"R2": 0, "MAPE": 0, "Acc": 0}
    r2 = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / np.where(y_true==0, 1e-10, y_true))) * 100
    acc = np.mean(np.sign(np.diff(y_true)) == np.sign(np.diff(y_pred))) * 100 if len(y_true) > 1 else 0
    return {"R2": r2, "MAPE": mape, "Acc": acc}

# 2. 투자 심의 위원회 LLM 로직
# ==========================================
def run_investment_committee(company, data_summary_str, api_key, scenario):
    if not api_key: return None
    client = OpenAI(api_key=api_key)
    
    agents = [
        {"role": "리스크 관리자 (보수적)", "icon": "🛡️", "prompt": "부채, 우발채무, 현금흐름의 질을 근거로 비판적 시각에서 리스크를 지적하세요."},
        {"role": "성장 분석가 (공격적)", "icon": "🚀", "prompt": "매출 성장성, 뉴스 모멘텀, 산업의 기회를 근거로 투자 가치를 옹호하세요."},
        {"role": "계량 분석가 (퀀트)", "icon": "🔢", "prompt": "ML 모델 지표, IS 마진율, 수치적 상관관계의 신뢰성을 엄격히 따지세요."}
    ]
    debate_log = []
    
    for agent in agents:
        prompt = f"""
        당신은 투자 심의 위원회의 {agent['role']}입니다. 
        대상 기업: {company} | 거시경제 상황: {scenario}
        데이터 요약: {data_summary_str}
        
        미션: {agent['prompt']}
        친근한 한국어 해요체로 3~4문장으로 발언하세요. JSON 형식으로 반환:
        {{ "comment": "발언 내용" }}
        """
        try:
            response = client.chat.completions.create(model="gpt-3.5-turbo", messages=[{"role": "user", "content": prompt}], response_format={"type": "json_object"})
            comment = json.loads(response.choices[0].message.content)['comment']
        except Exception as e:
            comment = f"분석 오류 발생: {e}"
        debate_log.append({"role": agent['role'], "icon": agent['icon'], "content": comment})

    prompt_chair = f"""
    당신은 투자 심의 위원회의 의장입니다. 아래 분석가들의 토론 내용을 바탕으로 {company}에 대한 최종 투자의견을 결정하세요.
    토론 내용: {debate_log}
    거시경제 상황: {scenario}
    
    JSON 형식으로만 반환:
    {{
        "decision": "Strong Buy / Buy / Hold / Sell 중 선택",
        "synthesis": "토론을 요약하고 내린 최종 결론 (해요체 4문장)",
        "score": 0~100 (투자 매력도 점수)
    }}
    """
    try:
        res_chair = json.loads(client.chat.completions.create(model="gpt-3.5-turbo", messages=[{"role": "user", "content": prompt_chair}], response_format={"type": "json_object"}).choices[0].message.content)
    except:
        res_chair = {"decision": "Error", "synthesis": "최종 의견을 도출하지 못했습니다.", "score": 0}
        
    return {"debate": debate_log, "final": res_chair}
